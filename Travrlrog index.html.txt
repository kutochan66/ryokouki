<script>
    // 初期設定
    const map = L.map('map', { zoomControl: false }).setView([35.68, 139.76], 5);
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

    let travelData = JSON.parse(localStorage.getItem('KutoseaTravelData')) || [];
    let currentPos = [35.6812, 139.7671];
    let routeLine = null;

    // 起動時の処理
    window.onload = () => {
        renderMarkers();
        setTimeout(() => {
            const splash = document.getElementById('splash');
            splash.style.opacity = '0';
            setTimeout(() => splash.style.display = 'none', 800);
        }, 2200);
    };

    // 写真から位置情報を取得
    document.getElementById('photoInput').addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) return;
        EXIF.getData(file, function() {
            const lat = EXIF.getTag(this, "GPSLatitude");
            const lon = EXIF.getTag(this, "GPSLongitude");
            if (lat && lon) {
                const formatCoord = (coord, ref) => {
                    let d = coord[0] + coord[1] / 60 + coord[2] / 3600;
                    return (ref === "S" || ref === "W") ? d * -1 : d;
                };
                currentPos = [formatCoord(lat, EXIF.getTag(this, "GPSLatitudeRef")), formatCoord(lon, EXIF.getTag(this, "GPSLongitudeRef"))];
                map.setView(currentPos, 15);
                document.getElementById('status').innerText = "📍 場所を特定しました！";
                document.getElementById('status').style.color = "#28a745";
            } else {
                document.getElementById('status').innerText = "⚠️ 位置情報が見つかりません";
                document.getElementById('status').style.color = "#dc3545";
            }
        });
    });

    function addRecord() {
        const amount = parseInt(document.getElementById('amount').value) || 0;
        const newEntry = {
            pos: currentPos,
            cat: document.getElementById('category').value,
            amt: amount,
            time: new Date().toLocaleString()
        };
        travelData.push(newEntry);
        localStorage.setItem('KutoseaTravelData', JSON.stringify(travelData));
        renderMarkers();
        document.getElementById('amount').value = "";
        alert("記録を保存しました！");
    }

    function renderMarkers() {
        let total = 0;
        map.eachLayer(l => { if (l instanceof L.Marker || l instanceof L.Polyline) map.removeLayer(l); });

        travelData.forEach(item => {
            total += item.amt;
            L.marker(item.pos).addTo(map).bindPopup(`<b>${item.cat}</b><br>${item.amt}円`);
        });
        document.getElementById('totalAmount').innerText = total.toLocaleString();
    }

    function playRoute() {
        if (travelData.length < 2) return alert("記録が2件以上必要です");
        if (routeLine) map.removeLayer(routeLine);
        routeLine = L.polyline([], {color: '#ffc107', weight: 5, dashArray: '10, 10'}).addTo(map);

        let i = 0;
        const interval = setInterval(() => {
            if (i >= travelData.length) { clearInterval(interval); return; }
            routeLine.addLatLng(travelData[i].pos);
            map.panTo(travelData[i].pos);
            i++;
        }, 1000);
    }

    function resetData() {
        if (confirm("全てのデータを消去してよろしいですか？")) {
            localStorage.removeItem('KutoseaTravelData');
            location.reload();
        }
    }
</script>
